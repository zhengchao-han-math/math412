\begin{align*} \vert \int_{c}^{d} c_{k}(x)\, dx - \int_{c}^{d} c_{l}(x)\, dx\vert \amp\le \int_{c}^{d} \vert c_{k}(x)-c_{l}(x)\vert \, dx\\ \amp\le \left(\int_{c}^{d} \vert c_{k}(x)-c_{l}(x)\vert^{p} \, dx\right)^{1/p}(d-c)^{1-1/p},\\ \left| \Vert c_{k} \Vert_{L^{q}[c,d]} -\Vert c_{l} \Vert_{L^{q}[c,d]}\right| \amp\le \Vert c_{k} -c_{l}\Vert_{L^{q}[c,d]}\\ \amp \le \Vert c_{k} -c_{l}\Vert_{L^{p}[c,d]}(d-c)^{1/q-1/p}. \end{align*}
